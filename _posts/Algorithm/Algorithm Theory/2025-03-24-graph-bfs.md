### 5.1. C++

```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

void bfs(int startNode, const vector<vector<int>>& graph) {
    vector<bool> visited(graph.size(), false);
    queue<int> q;

    // μ‹μ‘ λ…Έλ“λ¥Ό νμ— μ¶”κ°€ν•κ³  λ°©λ¬Έ μ²λ¦¬
    q.push(startNode);
    visited[startNode] = true;

    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";

        // μΈμ ‘ν• λ…Έλ“ νƒμƒ‰
        for (int neighbor : graph[node]) {
            if (!visited[neighbor]) {
                q.push(neighbor);
                visited[neighbor] = true; // λ°©λ¬Έ μ²λ¦¬
            }
        }
    }
}

int main() {
    // κ·Έλν”„ μ •μ (μ: 0-1-2-3-4)
    vector<vector<int>> graph = {
        {1, 2},    // 0
        {0, 3, 4}, // 1
        {0},       // 2
        {1},       // 3
        {1}        // 4
    };

    bfs(0, graph); // μ‹μ‘ λ…Έλ“: 0
    return 0;
}
```
